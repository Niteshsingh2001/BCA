                #include <stdio.h>
                #include <math.h>
                int main()
                {
                    int num,temp,power=0,totalSum = 0;
                    printf("Enter a number : ");
                    scanf("%d", &num);
                    temp = num;
                    while (temp != 0)
                    {
                        power++;
                        temp /= 10;
                    }
                    temp = num;
                    while (temp != 0)
                    {
                        int singleDigit = temp % 10;
                        totalSum += pow(singleDigit, power);
                        temp /= 10;
                    }
                    if (totalSum == num)
                        printf("This number is Armstrong number \n");
                    else
                        printf("This number is not Armstrong number \n");
                    return 0;
                }

                #include <stdio.h>

                int nCr(int N, int r)
                {
                    int res = 0;
                    if (r == 0)
                    {
                        return 1;
                    }
                    else
                    {
                        res = nCr(N, r - 1) * (N - r + 1) / r;
                    }
                    return res;
                }
